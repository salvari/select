%(BEGIN_ANSWER)

\medskip
\item{$\bullet$} 0$^{o}C$ = 32$^{o}F$
\item{$\bullet$} 100$^{o}C$ = 212$^{o}F$
\item{$\bullet$} -40$^{o}C$ = -40$^{o}F$
\item{$\bullet$} 250$^{o}C$ = 482$^{o}F$
\item{$\bullet$} 310$^{o}C$ = 590$^{o}F$
\medskip

$$T_C = {5 \over 9} \left(T_F - 32\right)$$
