%(BEGIN_ANSWER)

\medskip
\item{$\bullet$} $100110_2$ = $110101_{\hbox{Gray}}$
\vskip 5pt
\item{$\bullet$} $110010_2$ = $101011_{\hbox{Gray}}$
\vskip 5pt
\item{$\bullet$} $101001_2$ = $111101_{\hbox{Gray}}$
\vskip 5pt
\item{$\bullet$} $1010100110_2$ = $1111110101_{\hbox{Gray}}$
\medskip

%(END_ANSWER)
